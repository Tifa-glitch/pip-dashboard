// 蓓福業務日報 Dashboard — Apps Script（第三期 5–6月） // 試算表 ID：14GJSTNNjGfrmuU8tyGGSZEhICL8jMRUGIgle540ZhpA // ⚠ 每次修改後需「新增部署作業」才會生效 function doGet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); // ── 日報輸入 ── // 欄位對應（依截圖）： // A=月份 B=日期 C=業務姓名 D=訂單數 // E=中判陳列優化傳統 F=中判陳列優化連鎖 // G=大判陳列優化傳統 H=大判陳列優化連鎖 // I=超值包傳統家數 J=超值包傳統盒數 // K=超值包連鎖家數 L=超值包連鎖盒數 // M=今日收款金額（今日新增） N=累計收款金額（累計） O=備註 const dataSheet = ss.getSheetByName('日報輸入'); const dataRows = dataSheet.getDataRange().getValues(); const raw = []; for (let i = 2; i < dataRows.length; i++) { // 從第3列開始（跳過標題+說明列） const r = dataRows[i]; const nameVal = String(r[2] || '').trim(); // C欄 if (!nameVal || nameVal === '下拉選單' || nameVal.includes('填寫格式')) continue; // 月份（A欄） let monthKey = ''; const monthVal = r[0]; if (monthVal instanceof Date) { monthKey = monthVal.getFullYear() + '-' + String(monthVal.getMonth() + 1).padStart(2, '0'); } else { monthKey = String(monthVal || '').trim(); } // 日期（B欄） let dateKey = ''; const dateVal = r[1]; if (dateVal instanceof Date) { const m = String(dateVal.getMonth() + 1).padStart(2, '0'); const d = String(dateVal.getDate()).padStart(2, '0'); dateKey = m + '/' + d; } else { dateKey = String(dateVal || '').slice(5, 10).replace('-', '/'); } raw.push({ month : monthKey, date : dateKey, name : nameVal, order : Number(r[3]) || 0, // D 訂單數（今日新增） mdt : Number(r[4]) || 0, // E 中判陳列優化 傳統（累計） mdc : Number(r[5]) || 0, // F 中判陳列優化 連鎖（累計） bdt : Number(r[6]) || 0, // G 大判陳列優化 傳統（累計） bdc : Number(r[7]) || 0, // H 大判陳列優化 連鎖（累計） svt_f : Number(r[8]) || 0, // I 超值包鋪貨 傳統家數（累計） svt_b : Number(r[9]) || 0, // J 超值包鋪貨 傳統盒數（累計） svc_f : Number(r[10]) || 0, // K 超值包鋪貨 連鎖家數（累計） svc_b : Number(r[11]) || 0, // L 超值包鋪貨 連鎖盒數（累計） pay : Number(r[12]) || 0, // M 今日收款金額（今日新增） pay_cum : Number(r[13]) || 0, // N 累計收款金額（累計） }); } // ── 業務目標值 ── const tgtSheet = ss.getSheetByName('業務目標值'); const tgtRows = tgtSheet.getDataRange().getValues(); const targets = {}; for (let i = 1; i < tgtRows.length; i++) { const r = tgtRows[i]; const name = String(r[2] || '').trim(); // C欄＝業務姓名 if (!name) continue; targets[name] = { mdt : r[3] !== '' ? Number(r[3]) : null, mdc : r[4] !== '' ? Number(r[4]) : null, bdt : r[5] !== '' ? Number(r[5]) : null, bdc : r[6] !== '' ? Number(r[6]) : null, svt_f : r[7] !== '' ? Number(r[7]) : null, svt_b : r[8] !== '' ? Number(r[8]) : null, svc_f : r[9] !== '' ? Number(r[9]) : null, svc_b : r[10] !== '' ? Number(r[10]) : null, }; } return ContentService .createTextOutput(JSON.stringify({ raw, targets })) .setMimeType(ContentService.MimeType.JSON); }
